int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

int suma(int a, int b) {
    return a + b;
}

int main() {
    int x = 5;
    int y = 10;
    int resultado = suma(x, y);
    int fact = factorial(x);
    int i = 0;
    while (i < 5) {
        i++;
    }
    for (int j = 0; j < 3; j++) {
        resultado = resultado + j;
    }
    return 0;
}
