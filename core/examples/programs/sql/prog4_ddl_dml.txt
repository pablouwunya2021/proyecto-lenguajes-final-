CREATE TABLE empleados (
    id INTEGER PRIMARY KEY NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    salario INTEGER,
    dep_id INTEGER
) ;

CREATE TABLE departamentos (
    id INTEGER PRIMARY KEY NOT NULL,
    nombre VARCHAR(50) NOT NULL
) ;
