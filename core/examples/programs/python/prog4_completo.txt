class Calculadora:
    def init(valor):
        self.resultado = valor
    end

    def sumar(n):
        self.resultado = self.resultado + n
        return self.resultado
    end

    def restar(n):
        self.resultado = self.resultado - n
        return self.resultado
    end

    def multiplicar(n):
        self.resultado = self.resultado * n
        return self.resultado
    end

    def reset():
        self.resultado = 0
    end
end

def aplicar_lista(lista, factor):
    i = 0
    acum = 0
    while i < factor:
        acum = acum + lista
        i = i + 1
    end
    return acum
end

def buscar_max(coleccion, limite):
    maximo = 0
    for elem in coleccion:
        if elem > maximo:
            if elem < limite:
                maximo = elem
            end
        end
    end
    return maximo
end
